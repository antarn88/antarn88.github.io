Programozási tétel: Buborék rendezés (Bubble sort)
Növekvő sorba rendezés esetén.
Amennyiben csökkenőt szeretnénk, az IF-es sorban megcseréljük a relációs jelet.
A függvény vár egy szám tömböt, majd kiköpi ugyanazt a tömböt sorba rendezve.
public static int[] BubbleSort(int[] array)
{
 bool swapped;
 do
 {
  swapped = false;
  for (int i = 0; i < array.Length - 1; i++)
  {
   if (array[i] > array[i + 1])
   {
    int temp = array[i];
    array[i] = array[i + 1];
    array[i + 1] = temp;
    swapped = true;
   }
  }
 } while (swapped);
 return array;
}
Ezután meghívhatjuk ezt a Main-ben:
int[] numbers = { 1, 2, 333, 800, 77, -20 };
foreach (int number in BubbleSort(numbers))
{
 Console.WriteLine(number);
}

OOP - Konstruktor
Konstruktor létrehozása az egyedi osztályon belül paraméterekkel:
class Dog
{
 // Osztály attribútumok, amik jellemzik az adott osztályt
 public string name;
 public string color;
 public int age;

 // Konstruktor, ami minden példányosításkor lefut, de csak egyszer!
 // Zárójelben megadjuk, mik azok a paraméterek, amiket meg kell adni,
 // hogy a példány létrejöjjön a másik oldalon, ezeket várjuk el!
 public Dog(string name, string color, int age)
 {
  // Beállítjuk, hogy a felhasználó által megadott paraméterek legyenek
  // a példány tulajdonságai.
  this.name = name;
  this.color = color;
  this.age = age;
 }
}
A példányosítás például a Main-ben:
// Példányosítjuk az osztályt különböző paraméterekkel, tulajdonságokkal

// Majd lekérjük az osztály példányának megadott tulajdonságait
// Ez csak akkor működik, ha az adattagok publikusak!

// Mindenképpen be kell importálni legfelül!!!
using MySql.Data.MySqlClient;

// Osztályváltozók (ha OOP modellben gondolkodunk)
public string server = "localhost";
public string port = "3306";
public string username = "root";
public string password = "";
public string database = "test_db";

// A függvény vár különböző kapcsolati (osztály)változókat!
public MySqlConnection GetConnection()
{
 string connectionString = $"server={server};" +
              $"port={port};" +
              $"username={username};" +
              $"password={password};" +
              $"database={database};";

 MySqlConnection connection = new MySqlConnection(connectionString);

 try
 {
  connection.Open();
  return connection;
 }
 catch (Exception)

// A függvény vár egy GetConnection nevű MySqlConnection kapcsolatot, egy query-t,
// és opcionálisan egy oszlopszeparátor karaktert/stringet!
public List<string> GetDataRows(string query, string separator = "|")
{
 MySqlConnection connection = GetConnection();
 List<string> rows = new List<string>();

 if (connection != null)
 {
  try
  {
   MySqlCommand cmd = new MySqlCommand(query, connection);
   MySqlDataReader reader = cmd.ExecuteReader();
   int columnsCount = reader.FieldCount;

   if (reader.HasRows)
   {
    while (reader.Read())
    {
     string row = "";
     for (int i = 0; i < columnsCount; i++)
      row += i < columnsCount - 1 ? $"{reader.GetString(i)}{separator}" : reader.GetString(i);
     rows.Add(row);
    }
   }
   else
    Console.WriteLine("Nincsenek adatsorok.");
   connection.Close();
   return rows;
  }
  catch (Exception)
  {
   connection.Close();
   Console.WriteLine("Hiba történt az adatlekérés során!");
   return rows;
  }
 }
 else
 {
  Console.WriteLine("Connection hiba miatt a query nem lett végrehajtva!");
  return rows;
 }
}
